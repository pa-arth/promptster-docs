---
title: "List Sessions"
api: "GET /v1/sessions"
description: "List assessment sessions for your organization with pagination and filtering."
---

Returns a paginated list of sessions ordered by most recent activity. You can filter by source service (e.g. `claude_code`, `cursor`) or by assessment ID.

### Authentication

Requires either an org API key or a candidate key.

```
Authorization: Bearer psk_live_...
```

<ParamField query="limit" type="integer" default="20">
  Number of sessions to return. Maximum 100.
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Number of sessions to skip for pagination.
</ParamField>

<ParamField query="sourceService" type="string">
  Filter by source service (e.g. `claude_code`, `cursor`).
</ParamField>

<ParamField query="assessmentId" type="string">
  Filter to sessions linked to a specific assessment.
</ParamField>

<ResponseField name="sessions" type="array">
  List of session objects.
  <Expandable title="properties">
    <ResponseField name="id" type="string">Session ID.</ResponseField>
    <ResponseField name="orgId" type="string">Organization ID.</ResponseField>
    <ResponseField name="sourceService" type="string[]">Source services (e.g. `["claude_code"]`).</ResponseField>
    <ResponseField name="startedAt" type="string | null">When the session started.</ResponseField>
    <ResponseField name="endedAt" type="string | null">When the session ended.</ResponseField>
    <ResponseField name="lastActivityAt" type="string | null">Last event timestamp.</ResponseField>
    <ResponseField name="metadata" type="object">Session metadata.</ResponseField>
    <ResponseField name="eventCount" type="integer">Total number of timeline events.</ResponseField>
    <ResponseField name="decisionFlagCount" type="integer">Number of flagged decisions.</ResponseField>
    <ResponseField name="aiAttributionPct" type="number | null">Percentage of code attributed to AI.</ResponseField>
    <ResponseField name="totalCostUsd" type="number">Total LLM token cost in USD.</ResponseField>
    <ResponseField name="createdAt" type="string">ISO 8601 creation timestamp.</ResponseField>
    <ResponseField name="updatedAt" type="string">ISO 8601 last update timestamp.</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total" type="integer">
  Total number of matching sessions.
</ResponseField>

<ResponseField name="limit" type="integer">
  The limit used for this request.
</ResponseField>

<ResponseField name="offset" type="integer">
  The offset used for this request.
</ResponseField>
